DATA_SEG SEGMENT
    ARR DB 77H, 78H, 65H, 66H, 44H, 21H, 98H, 10H, 74H, 0FH
    EVEN_COUNT DB 0 
    ODD_COUNT DB 0
DATA_SEG ENDS

CODE_SEG SEGMENT
    ASSUME CS:CODE_SEG, DS:DATA_SEG
    
START:
    MOV AX, DATA_SEG
    MOV DS, AX

    LEA SI, ARR
    MOV CX, 10 
AGAIN: 
	MOV AL, [SI]           
	ROR AL, 1          
	JC ODDS
	INC EVEN_COUNT
	JMP NEXT 
ODDS: 
	INC ODD_COUNT
NEXT : 
	INC SI
        LOOP AGAIN

	MOV AH, 4CH
	INT 21H
    
CODE_SEG ENDS
END START
